# Day29 回溯5

## 递增子序列

需要去重的是同一层的相同数字

```go
func findSubsequences(nums []int) [][]int {
    var res [][]int
    var cur []int
    var dfs func(nums []int, index int)
    dfs = func(nums []int, index int){
        if len(cur) >1{
            tmp := make([]int, len(cur))
            copy(tmp, cur)
            res = append(res, tmp)
        }
        visited := make(map[int]bool, len(nums)) // 去重
        for i := index;i<len(nums);i++{
            if visited[nums[i]]{
                continue
            }
            if len(cur) > 0 && nums[i]< cur[len(cur)-1]{
                continue
            }
            if i > index && nums[i] == nums[i-1]{
                continue
            }
            visited[nums[i]] = true
            cur = append(cur, nums[i])
            dfs(nums, i+1)
            cur = cur[:len(cur)-1]
        }

    }
    dfs(nums,0)
    return res
}
```

## 全排列

```go
func permute(nums []int) [][]int {
    var res [][]int
    var cur []int
    var dfs func(nums []int)
    visited := make([]bool, len(nums))
    dfs = func(nums []int){
        if len(cur) == len(nums){
            tmp := make([]int, len(cur))
            copy(tmp, cur)
            res = append(res, tmp)
            return
        }
        for i := range nums{
            if visited[i]{
                continue
            }
            visited[i] = true
            cur = append(cur, nums[i])
            dfs(nums)
            visited[i] = false
            cur = cur[:len(cur)-1]
        }
    }
    dfs(nums)
    return res
}
```

## 全排列2

这次有重复数字且要求去重

```go
func permuteUnique(nums []int) [][]int {
    var res [][]int
    var cur []int
    sort.Ints(nums) // 别忘了排序
    visitied := make([]bool, len(nums))
    var dfs func(nums []int)
    dfs = func(nums []int){
        if len(cur) == len(nums){
            tmp := make([]int, len(cur))
            copy(tmp,cur)
            res = append(res, tmp)
            return
        }

        for i := range nums{
            if i > 0 && nums[i] == nums[i-1] && !visitied[i-1]{
                // 这个不能被选去
                continue
            }
            if visitied[i]{
                continue
            }
            visitied[i] = true
            cur = append(cur, nums[i])
            dfs(nums)
            cur = cur[:len(cur)-1]
            visitied[i] = false
        }

    }

    dfs(nums)
    return res
}
```
